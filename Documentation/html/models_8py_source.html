23 @staticmethod
24 def authorization(user_id, login):
25 '''! Поиск пользователя по логину, если не найден то присвоить логин текущему user_id.
26 В противном случае вернуть новый user_id
27 @param user_id int
28 @param login string
29 @return user_id int
30 '''
31 try:
32 user = User.objects.get(login=login)
33 user_id = user.pk
34 except User.DoesNotExist:
35 User.objects.filter(pk=user_id).update(login=login)
36 return user_id
37
38 @staticmethod
39 def get_user_id(request):
40 '''! Возвращаем id пользователя - залогиненного или выданного по session_id
41 @param request HttpRequest
42 @return user_id int
43 '''
44 if request.user.is_authenticated:
45 current_user = request.user
46 else:
47 current_user = User.objects.get(session_id=request.session.session_key)
48 return current_user.id
