52def string_to_minutes(string):
53 '''! Возвращает количество минут в строке

55 @param string string
56 @return int
57 '''
58 nums = [int(x) for x in re.findall(r'\d+', string)]

60 if 'мин' in string:

63 return nums[0] * 60
